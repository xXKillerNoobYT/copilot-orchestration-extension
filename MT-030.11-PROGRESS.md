document.getElementById( 'submit-test-btn')?.addEventListener('click', async () => {
    const query = (document.getElementById('test-query') as HTMLTextAreaElement).value.trim();
    
    if (!query) {
        document.getElementById('test-error-text').textContent = 'âŒ Please enter a query';
        document.getElementById('test-error-section').style.display = 'block';
        return;
    }
    
    // Show status, hide results
    document.getElementById('test-status').style.display = 'flex';
    document.getElementById('test-output-section').style.display = 'none';
    document.getElementById('test-error-section').style.display = 'none';
    
    const startTime = performance.now();
    
    try {
        // Send test request to extension
        const result = await new Promise((resolve, reject) => {
            const handler = (event: MessageEvent) => {
                if (event.data.type === 'testResult') {
                    window.removeEventListener('message', handler);
                    if (event.data.error) {
                        reject(new Error(event.data.error));
                    } else {
                        resolve(event.data);
                    }
                }
            };
            
            window.addEventListener('message', handler);
            
            vscode.postMessage({
                type: 'testAgent',
                agentConfig: currentAgent,
                query: query
            });
            
            setTimeout(() => reject(new Error('Test timeout (30s)')), 30000);
        });
        
        const endTime = performance.now();
        const responseTime = Math.round(endTime - startTime);
        
        // Display results
        document.getElementById('test-response').textContent = result.response || '(No response)';
        document.getElementById('prompt-tokens').textContent = String(result.tokens?.prompt_tokens || 0);
        document.getElementById('completion-tokens').textContent = String(result.tokens?.completion_tokens || 0);
        document.getElementById('total-tokens').textContent = String(result.tokens?.total_tokens || 0);
        document.getElementById('response-time').textContent = String(responseTime);
        
        document.getElementById('test-status').style.display = 'none';
        document.getElementById('test-output-section').style.display = 'block';
        
    } catch (error) {
        const errorMsg = error instanceof Error ? error.message : 'Unknown error';
        document.getElementById('test-error-text').textContent = `âŒ ${errorMsg}`;
        document.getElementById('test-status').style.display = 'none';
        document.getElementById('test-error-section').style.display = 'block';
    }
});
