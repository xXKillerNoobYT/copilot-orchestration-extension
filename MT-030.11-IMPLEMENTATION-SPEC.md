### 3. Add JavaScript (in getJavaScript, setupEventListeners)
```typescript
// Test agent button
document.getElementById('test-agent-btn')?.addEventListener('click', () => {
    const panel = document.getElementById('test-mode-panel');
    panel.style.display = 'block';
    document.getElementById('test-query').focus();
});

document.getElementById('close-test-btn')?.addEventListener('click', () => {
    document.getElementById('test-mode-panel').style.display = 'none';
});

document.getElementById('submit-test-btn')?.addEventListener('click', async () => {
    const query = (document.getElementById('test-query') as HTMLTextAreaElement).value;
    
    if (!query.trim()) {
        document.getElementById('test-error-text').textContent = 'Please enter a query';
        document.getElementById('test-error-section').style.display = 'block';
        return;
    }
    
    document.getElementById('test-status').style.display = 'flex';
    document.getElementById('test-output-section').style.display = 'none';
    document.getElementById('test-error-section').style.display = 'none';
    
    try {
        const startTime = performance.now();
        
        const response = await new Promise((resolve, reject) => {
            const listener = (event) => {
                if (event.data.type === 'testResult') {
                    window.removeEventListener('message', listener);
                    if (event.data.error) reject(new Error(event.data.error));
                    else resolve(event.data);
                }
            };
            
            window.addEventListener('message', listener);
            
            vscode.postMessage({
                type: 'testAgent',
                agentConfig: currentAgent,
                testQuery: query
            });
            
            setTimeout(() => reject(new Error('Test timeout (30s)')), 30000);
        });
        
        const endTime = performance.now();
        const responseTime = Math.round(endTime - startTime);
        
        document.getElementById('test-response').textContent = response.response;
        document.getElementById('prompt-tokens').textContent = response.usage?.prompt_tokens || 0;
        document.getElementById('completion-tokens').textContent = response.usage?.completion_tokens || 0;
        document.getElementById('total-tokens').textContent = response.usage?.total_tokens || 0;
        document.getElementById('response-time').textContent = responseTime;
        
        document.getElementById('test-output-section').style.display = 'block';
    } catch (error) {
        document.getElementById('test-error-text').textContent = error instanceof Error ? error.message : 'Unknown error';
        document.getElementById('test-error-section').style.display = 'block';
    } finally {
        document.getElementById('test-status').style.display = 'none';
    }
});
```

### 4. Add Message Handler in Extension (src/ui/customAgentBuilder.ts)
In the panel message listener, add:
```typescript
case 'testResult': {
    // Forward to webview
    this.panel.webview.postMessage(message);
    break;
}
```

### 5. Add Handler in Extension Main (src/extension.ts or orchestrator)
When webview sends 'testAgent':
```typescript
case 'testAgent': {
    const agentConfig = message.agentConfig;
    const testQuery = message.testQuery;
    
    try {
        // Execute agent with test query
        const result = await agentExecutor.execute(agentConfig, testQuery);
        
        panel.webview.postMessage({
            type: 'testResult',
            response: result.response,
            usage: result.tokens
        });
    } catch (error) {
        panel.webview.postMessage({
            type: 'testResult',
            error: error.message
        });
    }
    break;
}
```
