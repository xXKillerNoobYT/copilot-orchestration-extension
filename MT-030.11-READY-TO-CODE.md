// Test agent button
document.getElementById('btn-test')?.addEventListener('click', () => {
    const modal = document.getElementById('test-mode-modal');
    if (modal) modal.style.display = 'flex';
    const queryInput = document.getElementById('test-query') as HTMLTextAreaElement;
    if (queryInput) queryInput.focus();
});

document.getElementById('close-test-modal')?.addEventListener('click', () => {
    const modal = document.getElementById('test-mode-modal');
    if (modal) modal.style.display = 'none';
});

document.getElementById('submit-test-btn')?.addEventListener('click', async () => {
    const queryInput = document.getElementById('test-query') as HTMLTextAreaElement;
    const query = queryInput ? queryInput.value.trim() : '';
    
    if (!query) {
        const errorText = document.getElementById('test-error-text');
        if (errorText) errorText.textContent = '❌ Please enter a query';
        const errorSection = document.getElementById('test-error-section');
        if (errorSection) errorSection.style.display = 'block';
        return;
    }
    
    const statusEl = document.getElementById('test-status');
    const outputEl = document.getElementById('test-output-section');
    const errorEl = document.getElementById('test-error-section');
    
    if (statusEl) statusEl.style.display = 'flex';
    if (outputEl) outputEl.style.display = 'none';
    if (errorEl) errorEl.style.display = 'none';
    
    const startTime = performance.now();
    
    try {
        const result = await new Promise<any>((resolve, reject) => {
            const handler = (event: MessageEvent) => {
                if (event.data.type === 'testResult') {
                    window.removeEventListener('message', handler);
                    if (event.data.error) {
                        reject(new Error(event.data.error));
                    } else {
                        resolve(event.data);
                    }
                }
            };
            
            window.addEventListener('message', handler);
            vscode.postMessage({
                type: 'testAgent',
                agentConfig: currentAgent,
                query: query
            });
            
            setTimeout(() => reject(new Error('Test timeout (30s)')), 30000);
        });
        
        const endTime = performance.now();
        const responseTime = Math.round(endTime - startTime);
        
        const responseEl = document.getElementById('test-response');
        if (responseEl) responseEl.textContent = result.response || '(No response)';
        
        const promptEl = document.getElementById('prompt-tokens');
        if (promptEl) promptEl.textContent = String(result.tokens?.prompt_tokens || 0);
        
        const completionEl = document.getElementById('completion-tokens');
        if (completionEl) completionEl.textContent = String(result.tokens?.completion_tokens || 0);
        
        const totalEl = document.getElementById('total-tokens');
        if (totalEl) totalEl.textContent = String(result.tokens?.total_tokens || 0);
        
        const timeEl = document.getElementById('response-time');
        if (timeEl) timeEl.textContent = String(responseTime);
        
        if (statusEl) statusEl.style.display = 'none';
        if (outputEl) outputEl.style.display = 'block';
        
    } catch (error) {
        const errorMsg = error instanceof Error ? error.message : 'Unknown error';
        const errorText = document.getElementById('test-error-text');
        if (errorText) errorText.textContent = '❌ ' + errorMsg;
        
        if (statusEl) statusEl.style.display = 'none';
        if (errorEl) errorEl.style.display = 'block';
    }
});

// Close modal on backdrop click
const modal = document.getElementById('test-mode-modal');
if (modal) {
    modal.addEventListener('click', (e) => {
        if ((e.target as HTMLElement).id === 'test-mode-modal') {
            modal.style.display = 'none';
        }
    });
}
